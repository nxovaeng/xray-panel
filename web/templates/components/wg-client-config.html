WireGuard 客户端配置参考

由于服务端只保存了客户端的公钥，以下配置中的 <客户端私钥> 需要您自行替换为您生成客户端公钥时对应的私钥。 

[Interface]
PrivateKey = <客户端私钥>
Address = {{.PeerIP}}
DNS = 1.1.1.1

[Peer]
PublicKey = {{.Inbound.WGPublicKey}}
Endpoint = {{.ServerIP}}:{{.Inbound.Port}}
AllowedIPs = 0.0.0.0/0, ::/0
PersistentKeepalive = 25
{{if gt .Inbound.WGMTU 0}}MTU = {{.Inbound.WGMTU}}{{end}}
{{end}}
